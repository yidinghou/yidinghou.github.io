pros_list = ["youdacao","Awesemo","papagates","mazwa", "ehafner","moklovin", "shawnzhan", "REDCOAT85","ChipotleAddict"]
i=0
plt.figure(figsize=(20,10))
plt.grid(True)
for pro in pros_list:
    Q = """
    SELECT * FROM CONTEST_DATA WHERE CONTESTID IN (SELECT CONTESTID from CONTEST_TYPE WHERE Year = 2018) 
    AND username ='%s'
    """%pro
    df = pd.read_sql(Q, NFL_conn)
    df = df[~df["ContestID"].isin(cashid)]
    group = df.groupby("Week")["Profit"].sum()
    plt.plot(group.cumsum(),"C"+str(i),label = pro)
    i=i+1
